### 2.1 单元测试

| 模块 | 测试范围 | 工具 |
|------|---------|------|
| **LayoutEngine** | 位置计算、碰撞检测、布局算法 | Google Test / xUnit |

| **FileWatcher** | 文件变更通知、事件解析 | Mock + Integration |

**示例测试用例：**

```csharp
[Theory]
[InlineData(100, 100, 300, 200, 350, 150, 400, 300)]  // 无重叠
[InlineData(100, 100, 300, 200, 200, 150, 400, 300)]   // 有重叠
public void DetectOverlap_ReturnsExpectedResult(int x1, int y1, int w1, int h1, 
    int x2, int y2, int w2, int h2, bool expected)
{
    var rect1 = new Rectangle(x1, y1, w1, h1);
    var rect2 = new Rectangle(x2, y2, w2, h2);
    
    var result = LayoutEngine.DetectOverlap(rect1, rect2);
    
    Assert.Equal(expected, result);
}

[Theory]
[InlineData(".exe", "C:\\App\\app.exe", true)]
[InlineData(".lnk", "C:\\Docs\\readme.lnk", true)]
[InlineData(".txt", "C:\\Docs\\readme.txt", false)]
public void FileExtensionRule_MatchesCorrectly(string extension, 
    string filePath, bool expected)
{
    var rule = new FileExtensionRule(extension);
    var item = new FenceItem { Path = filePath };
    
    var result = rule.Matches(item);
    
    Assert.Equal(expected, result);
}
```

**性能基准测试：**

```csharp
[Fact]
public void RenderFence_With500Icons_CompletesWithin16ms()
{
    // Arrange
    var fence = CreateFenceWithItems(500);
    var stopwatch = Stopwatch.StartNew();
    
    // Act
    fence.Render();
    
    // Assert
    stopwatch.Stop();
    Assert.True(stopwatch.ElapsedMilliseconds < 16, 
        $"渲染时间 {stopwatch.ElapsedMilliseconds}ms 超过 16ms");
}

[Fact]
public void MemoryUsage_IdleState_Under100MB()
{
    // Arrange
    StartApplication();
    WaitForIdle();
    
    // Act
    var memoryBefore = GetProcessMemory();
    Wait(TimeSpan.FromMinutes(5));
    var memoryAfter = GetProcessMemory();
    
    // Assert
    Assert.True(memoryAfter < 100 * 1024 * 1024, 
        $"内存占用 {memoryAfter / 1024 / 1024}MB 超过 100MB");
}
```
